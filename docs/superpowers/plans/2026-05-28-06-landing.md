import type { Station } from '@stray/ui';

const API = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000';

export async function fetchStations(city?: string): Promise<Station[]> {
  const url = city ? `${API}/stations?city=${encodeURIComponent(city)}` : `${API}/stations`;
  const res = await fetch(url, { next: { revalidate: 30 } });
  if (!res.ok) return [];
  return res.json();
}

export async function fetchStats(): Promise<{ total_donations: number; total_stations: number; total_events: number }> {
  // Derive stats from stations list — avoids a dedicated stats endpoint for now
  const stations = await fetchStations();
  return {
    total_donations: 0,  // Filled when donation totals endpoint exists
    total_stations: stations.filter(s => s.status !== 'offline').length,
    total_events: 0,
  };
}

'use client';
import { useEffect, useState } from 'react';
import { Eyebrow } from '@stray/ui';

interface Stats { online: number; cities: number; meals: number }

export function ImpactStrip() {
  const [stats, setStats] = useState<Stats>({ online: 0, cities: 4, meals: 184000 });

  useEffect(() => {
    const API = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000';
    const poll = async () => {
      try {
        const res = await fetch(`${API}/stations`);
        if (!res.ok) return;
        const stations = await res.json();
        setStats(prev => ({ ...prev, online: stations.filter((s: { status: string }) => s.status === 'online').length }));
      } catch { /* keep previous */ }
    };
    poll();
    const interval = setInterval(poll, 30000);
    return () => clearInterval(interval);
  }, []);

  const items = [
    { v: `${stats.online}`, l: 'Stations online now' },
    { v: `NT$2.75M`, l: 'Total raised' },
    { v: `${stats.cities}`, l: 'Cities live' },
    { v: '184k+', l: 'Meals funded' },
  ];

  return (
    <section id="impact" style={{ background: '#0f172a', padding: '72px 0' }}>
      <div style={{ maxWidth: 1240, margin: '0 auto', padding: '0 40px', display: 'grid', gridTemplateColumns: 'repeat(4,1fr)', gap: 40 }}>
        {items.map(item => (
          <div key={item.l} style={{ textAlign: 'center' }}>
            <div style={{ fontFamily: 'var(--font-sans)', fontWeight: 900, fontSize: 48, color: '#f97316', letterSpacing: '-0.03em', lineHeight: 1 }}>{item.v}</div>
            <Eyebrow color="rgba(255,255,255,0.5)" style={{ marginTop: 8 }}>{item.l}</Eyebrow>
          </div>
        ))}
      </div>
    </section>
  );
}
