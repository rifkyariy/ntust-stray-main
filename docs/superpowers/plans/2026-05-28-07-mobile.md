const API = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000';

export async function fetchStations(): Promise<Station[]> {

export async function fetchStation(id: string): Promise<Station | null> {
  const res = await fetch(`${API}/stations/${id}`, { cache: 'no-store' });
  if (!res.ok) return null;
  return res.json();
}

export async function postDonation(payload: {
  station_id: string;
  amount_ntd: number;
  donor_name?: string;
  dispense: boolean;
}): Promise<Donation | null> {
  const res = await fetch(`${API}/donations`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  });
  if (!res.ok) return null;
  return res.json();
}

  const res = await fetch(`${API}/stations/${id}/metrics?range=${range}`, { cache: 'no-store' });
  if (!res.ok) return [];
  return res.json();
}

```typescript
'use client';
import { useEffect, useRef, useCallback } from 'react';
import type { WSMessage } from '@stray/ui';

type Handler = (msg: WSMessage) => void;

export function useWebSocket(onMessage: Handler) {
  const wsRef = useRef<WebSocket | null>(null);
  const handlerRef = useRef<Handler>(onMessage);
  handlerRef.current = onMessage;

  const connect = useCallback(() => {
    const url = process.env.NEXT_PUBLIC_WS_URL ?? 'ws://localhost:8000/ws';
    const ws = new WebSocket(url);

    ws.onmessage = (event) => {
      try {
        const msg: WSMessage = JSON.parse(event.data);
        handlerRef.current(msg);
      } catch {
        // ignore malformed messages
      }
    };

    ws.onclose = () => {
      // Reconnect after 3s
      setTimeout(connect, 3000);
    };

    ws.onerror = () => {
      ws.close();
    };

    wsRef.current = ws;
  }, []);

  useEffect(() => {
    connect();
    return () => {
      wsRef.current?.close();
    };
  }, [connect]);
}
```

'use client';
import { useState } from 'react';
import { useWebSocket } from '../hooks/useWebSocket';
import { DetectionFeed } from './DetectionFeed';
import { SupplyLevels } from './SupplyLevels';
import type { Station, WSMessage, WSDetection, WSTelemetry } from '@stray/ui';
import { Wifi, Clock } from '@stray/ui';

export function StationDetailClient({ initialStation }: { initialStation: Station }) {
  const [station, setStation] = useState(initialStation);
  const [latestDetection, setLatestDetection] = useState<WSDetection | null>(null);

  useWebSocket((msg: WSMessage) => {
    if (msg.station_id !== initialStation.station_code) return;
    if (msg.type === 'telemetry') {
      const t = msg as WSTelemetry;
      setStation(prev => ({
        ...prev,
        food_pct: t.food_pct,
        battery_pct: t.battery_pct,
        temp_c: t.temp_c,
        humidity_pct: t.humidity_pct,
      }));
    }
    if (msg.type === 'detection') {
      setLatestDetection(msg as WSDetection);
    }
  });

  return (
    <div style={{ background: '#f8fafc', minHeight: '100vh', paddingBottom: 48 }}>
      <DetectionFeed station={station} latestDetection={latestDetection} />

      <div style={{ padding: '20px 16px 0' }}>
        {/* Meta pills */}
        <div style={{ display: 'flex', gap: 8, marginBottom: 20, flexWrap: 'wrap' }}>
          {[
            { icon: Wifi, label: station.station_code },
