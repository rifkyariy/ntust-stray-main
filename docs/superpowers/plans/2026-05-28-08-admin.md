# Plan 08 — Admin Console (apps/admin)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `apps/admin` Next.js auth-protected console — login page, AdminShell layout, real-time dashboard (KPI strip, SVG map, activity feed, stations table, station drawer), and secondary pages (cats, funding, alerts).

**Architecture:** Next.js 14 App Router. `middleware.ts` guards all routes under `/(dashboard)` by reading a `stray_admin_token` httpOnly cookie. **Tech Stack:** Next.js 14 App Router, React 18, @stray/ui, lucide-react, WebSocket API, js-cookie (client-side token read for WS auth header)

**Prerequisite:** Plans 01–05 complete.

import { NextRequest, NextResponse } from 'next/server';

const PUBLIC_PATHS = ['/login'];

export function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;
  if (PUBLIC_PATHS.some((p) => pathname.startsWith(p))) {
    return NextResponse.next();
  }
  const token = req.cookies.get('stray_admin_token')?.value;
  if (!token) {
    const loginUrl = req.nextUrl.clone();
    loginUrl.pathname = '/login';
    return NextResponse.redirect(loginUrl);
  }
  return NextResponse.next();
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico).*)'],
};
```

'use server';
import { cookies } from 'next/headers';

const COOKIE = 'stray_admin_token';
const MAX_AGE = 60 * 60 * 24; // 24 h

export function setAuthCookie(token: string) {
  cookies().set(COOKIE, token, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'lax',
    maxAge: MAX_AGE,
    path: '/',
  });
}

export function clearAuthCookie() {
  cookies().delete(COOKIE);
}

export function getAuthToken(): string | undefined {
  return cookies().get(COOKIE)?.value;
}

import type { Station, Donation } from '@stray/ui';

const API = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000';

// ── authenticated fetch (server components pass token explicitly) ──────────
export async function apiFetch(
  path: string,
  token: string,
  init: RequestInit = {},
): Promise<Response> {
  return fetch(`${API}${path}`, {
    ...init,
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${token}`,
      ...(init.headers ?? {}),
    },
    cache: 'no-store',
  });
}

  return fetch(`${API}${path}`, { cache: 'no-store' });

'use client';
import { useEffect, useRef, useCallback } from 'react';
import type { WSMessage } from '@stray/ui';

const WS_URL = process.env.NEXT_PUBLIC_WS_URL ?? 'ws://localhost:8000/ws';
const RECONNECT_DELAY_MS = 3000;

export type MessageHandler = (msg: WSMessage) => void;

export function useWebSocket(onMessage: MessageHandler) {
  const wsRef = useRef<WebSocket | null>(null);
  const handlerRef = useRef<MessageHandler>(onMessage);
  const deadRef = useRef(false);

  // keep handler ref fresh without re-connecting
  useEffect(() => {
    handlerRef.current = onMessage;
  }, [onMessage]);

  const connect = useCallback(() => {
    if (deadRef.current) return;
    const ws = new WebSocket(WS_URL);
    wsRef.current = ws;

    ws.onmessage = (ev) => {
      try {
        const msg: WSMessage = JSON.parse(ev.data);
        handlerRef.current(msg);
      } catch {
        // ignore malformed frames
      }
    };

    ws.onclose = () => {
      if (!deadRef.current) {
        setTimeout(connect, RECONNECT_DELAY_MS);
      }
    };

    ws.onerror = () => ws.close();
  }, []);

  useEffect(() => {
    deadRef.current = false;
    connect();
    return () => {
      deadRef.current = true;
      wsRef.current?.close();
    };
  }, [connect]);
}
```

import { useWebSocket } from '../../hooks/useWebSocket';
import type { Station, WSMessage } from '@stray/ui';
import type { KPIData } from '../../lib/api';

export default function DashboardPage() {
  const [stations, setStations] = useState<Station[]>([]);
  const [kpis, setKpis] = useState<KPIData>({
    stations_online: 0, dispensed_today_kg: 0, donated_today_ntd: 0,
    cats_tracked: 0, active_alerts: 0,
  });
  const [city, setCity] = useState('All cities');
  const [search, setSearch] = useState('');
  const [drawerStation, setDrawerStation] = useState<Station | null>(null);
  const [showAddModal, setShowAddModal] = useState(false);
  // token is read from cookie at runtime (server action not available in pure client component)
  // For drawer actions we read it from a meta tag set in the layout or pass via props
  const [token, setToken] = useState('');

  useEffect(() => {
    fetchStations().then((data) => {
      setStations(data);
      setKpis(deriveKPIs(data));
    });
    // Read token from cookie client-side (non-httpOnly for WS use — set separately by login)
    const match = document.cookie.match(/stray_admin_token=([^;]+)/);
    if (match) setToken(match[1]);
  }, []);

  const onMessage = useCallback((msg: WSMessage) => {
    if (msg.type === 'feed_event') {
      setKpis((prev) => ({
        ...prev,
        dispensed_today_kg: prev.dispensed_today_kg + (msg.grams / 1000),
        donated_today_ntd: prev.donated_today_ntd + 15, // NT$15 per dispense
      }));
    }
    if (msg.type === 'detection') {
      // track unique cats via station_id as rough count
      setKpis((prev) => ({ ...prev, cats_tracked: prev.cats_tracked + 1 }));
    }
  }, []);

  useWebSocket(onMessage);

  const filtered = stations.filter((s) => {
    const matchCity = city === 'All cities' || s.city === city;
    const q = search.toLowerCase();
    const matchSearch = !q || s.name.toLowerCase().includes(q) || s.station_code.toLowerCase().includes(q);
    return matchCity && matchSearch;
  });

  return (
    <>
      <Topbar
        city={city}
        onCityChange={setCity}
        onSearch={setSearch}
        onAddStation={() => setShowAddModal(true)}
        alertCount={kpis.active_alerts}
      />

      <div style={{ padding: '24px 28px' }}>
        <KPIStrip data={kpis} />

        <div style={{
          display: 'grid',
          gridTemplateColumns: '1fr 320px',
          gap: 20,
          marginBottom: 24,
          alignItems: 'start',
        }}>
          <StationsMap stations={filtered} onPinClick={setDrawerStation} />
          <ActivityFeed />

          token={token}

        />

          token={token}
          onClose={() => setShowAddModal(false)}
          onCreated={(s) => setStations((prev) => [...prev, s])}
        />
      )}
    </>
  );
}

import { useState, useCallback } from 'react';
import { ArrowLeft, Zap, Calendar, Thermometer, Droplets } from 'lucide-react';
import Link from 'next/link';
import { ProgressRow, MetricTile, StatusDot, TintedPill } from '@stray/ui';

import type { Station, WSMessage } from '@stray/ui';
import { PawPrint, Battery } from 'lucide-react';

const GRAM_PRESETS = [50, 100, 150, 200];

interface Props {
  station: Station;
  token: string;
}

export default function StationDrawerPage({ station: initial, token }: Props) {
  const [station, setStation] = useState(initial);
  const [dispenseGrams, setDispenseGrams] = useState(100);
  const [dispenseLoading, setDispenseLoading] = useState(false);
  const [dispenseSuccess, setDispenseSuccess] = useState(false);
  const [schedTime, setSchedTime] = useState('07:00');
  const [schedGrams, setSchedGrams] = useState(100);
  const [schedLoading, setSchedLoading] = useState(false);
  const [schedSuccess, setSchedSuccess] = useState(false);

  const onMessage = useCallback((msg: WSMessage) => {
    if (msg.type === 'telemetry' && msg.station_id === station.station_code) {
      setStation((prev) => ({
        ...prev,
        food_pct: msg.food_pct, battery_pct: msg.battery_pct,
        temp_c: msg.temp_c, humidity_pct: msg.humidity_pct,
        status: msg.food_pct < 20 ? 'low_food' : 'online',
      }));
    }
  }, [station.station_code]);

  useWebSocket(onMessage);

  async function handleDispense() {
    setDispenseLoading(true);
    const ok = await postDispense(station.id, dispenseGrams, token);
    setDispenseLoading(false);
    if (ok) setDispenseSuccess(true);
  }

  async function handleSchedule() {
    setSchedLoading(true);
    const [hh, mm] = schedTime.split(':');
    const ok = await postSchedule(station.id, `0 ${mm} ${hh} * * *`, schedGrams, token);
    setSchedLoading(false);
    if (ok) setSchedSuccess(true);

    : station.status === 'low_food' ? '#eab308' : '#94a3b8';

      <Link href="/" style={{
        display: 'inline-flex', alignItems: 'center', gap: 6,
        fontSize: 13, color: 'var(--slate-500)', textDecoration: 'none', marginBottom: 20,

        <ArrowLeft size={14} strokeWidth={2} /> Back to dashboard
      </Link>

      <div style={{ display: 'flex', alignItems: 'center', gap: 12, marginBottom: 28 }}>

            {station.station_code} · {station.city}, {station.district}
          </div>
        </div>
        <div style={{ marginLeft: 'auto' }}>
          <TintedPill
            variant={station.status === 'online' ? 'green' : station.status === 'low_food' ? 'orange' : 'slate'}
            label={station.status === 'online' ? 'Online' : station.status === 'low_food' ? 'Low food' : 'Offline'}
          />
        </div>
      </div>

      <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr 1fr', gap: 20 }}>
        {/* Camera */}
        <div style={{
          gridColumn: '1 / 3',
          background: 'var(--slate-900)', borderRadius: 16, height: 280,
          display: 'flex', alignItems: 'center', justifyContent: 'center',
          position: 'relative', overflow: 'hidden',
        }}>
          <span style={{ color: 'rgba(255,255,255,0.3)', fontSize: 14 }}>Live feed</span>

            borderRadius: 6, padding: '3px 8px',
            fontSize: 11, fontWeight: 700, letterSpacing: 1,
          }}>
            ● LIVE
          </div>
        </div>

            label="Temperature"
            color="var(--orange-500)"
            bg="var(--orange-50)"
          />

                width: '100%', padding: '10px 0',

            </div>

                fontSize: 13, fontFamily: 'var(--font-mono)', outline: 'none', color: 'var(--slate-900)',
